## តើធាតុ/ប្លុកកសាងអ្វីខ្លះចាំបាច់សម្រាប់អនុវត្តលំនាំរចនាប្រើប្រាស់ឧបករណ៍?

ប្លុកកសាងទាំងនេះអនុញ្ញាតឲ្យភ្នាក់ងារប្រើប្រាស់ AI អាចអនុវត្តបេសកកម្មជាច្រើន។ យើងនឹងមើលធាតុសំខាន់ៗចាំបាច់សម្រាប់អនុវត្តលំនាំរចនាប្រើប្រាស់ឧបករណ៍៖

- **ស្គីមា Function/Tool**៖ ការបកស្រាយលម្អិតពីឧបករណ៍មាន នៅជាមួយឈ្មោះអនុគមន៍ គោលបំណង ប៉ារ៉ាម៉ែត្រដែលត្រូវការ និងលទ្ធផលដែលរំពឹងទុក។ ស្គីមាទាំងនេះអនុញ្ញាតឲ្យម៉ូឌែលភាសាធំយល់ពីឧបករណ៍ដែលមាន និងវិធីសាស្ត្រតំរូវការត្រឹមត្រូវ។

- **តន្ត្រីអនុវត្តន៍ Function**៖ គ្រប់គ្រងរបៀបនិងពេលវេលាអោយឧបករណ៍ត្រូវបានហៅដោយផ្អែកលើចំណង់ចំណូលចិត្តរបស់អ្នកប្រើ និងបរិបទសន្ទនា។ វាអាចរួមបញ្ចូលម៉ូឌុលរៀបចំផែនការ ប្រព័ន្ធផ្ញើរទិស ឬលំនាំខ្សែទីផ្សារដែលសម្រេចចិត្តប្រើប្រាស់ឧបករណ៍យ៉ាងديناميكي។

- **ប្រព័ន្ធគ្រប់គ្រងសារសន្ទនា**៖ ធាតុខ្លះដែលគ្រប់គ្រងដំណើរការសន្ទនារវាងការបញ្ចូលពីអ្នកប្រើ ការឆ្លើយតបពីម៉ូឌែល ភូនៅអនុគមន៍ និងលទ្ធផលឧបករណ៍។

- **ស៊ុមរួមឧបករណ៍**៖ អាគារចោលដែលភ្ជាប់ភ្នាក់ងារជាមួយឧបករណ៍ផ្សេងៗ មិនថាជាអនុគមន៍សាមញ្ញ ឬសេវាកម្មក្រៅស្មុគស្មាញ។

- **ការគ្រប់គ្រងកំហុស និងផ្ទៀងផ្ទាត់**៖ ប្រព័ន្ធដែលដោះស្រាយករណីបរាជ័យក្នុងការអនុវត្តឧបករណ៍ ផ្ទៀងផ្ទាត់ប៉ារ៉ាម៉ែត្រ និងគ្រប់គ្រងចម្លើយមិនរំពឹងទុក។

- **ការគ្រប់គ្រងស្ថានភាព**៖ តាមដានបរិបទសន្ទនា ភាពងាយស្រួលឧបករណ៍មុន និងទិន្នន័យជារឿយៗដើម្បីធានាថាមានភាពចំរូងគ្នាជាប់ជាពេលវេលាច្រើនជំហាន។

ការហៅអនុគមន៍ជាវិធីសំខាន់ក្នុងការអនុញ្ញាតឲ្យម៉ូឌែលភាសាធំ (LLMs) អាចអន្តរាក់ជាមួយឧបករណ៍។ អ្នកតែងតែឃើញពាក្យ 'Function' និង 'Tool' ប្រើជំនួសគ្នាដោយសារតែ 'function' (ប្លុកកូដដែលអាចប្រើប្រាស់ឡើងវិញ) គឺជាឧបករណ៍ដែលភ្នាក់ងារប្រើអនុវត្តបេសកកម្ម។ ដើម្បីអនុវត្តបណ្តូរកូដអនុគមន៍ មួយ LLM ត្រូវតែប្រៀបធៀបសំណើរបស់អ្នកប្រើជាមួយការពិពណ៌នាអនុគមន៍។ ដើម្បីធ្វើការនេះ ស្គីមដែលមានការពិពណ៌នាពីអនុគមន៍ទាំងអស់ដែលមានត្រូវបានផ្ញើទៅ LLM។ LLM នោះជ្រើសរើសអនុគមន៍សមរម្យបំផុតសម្រាប់បេសកកម្មហើយត្រឡប់ឈ្មោះឡើងវិញជាមួយអាគុយម៉ង់។ អនុគមន៍ដែលបានជ្រើសរើសត្រូវបានហៅ ប្រាប់ត្រឡប់តបនឹង LLM ដែលប្រើព័ត៌មាននេះឆ្លើយតបសំណើអ្នកប្រើ។

សម្រាប់អ្នកអភិវឌ្ឍដែលចង់អនុវត្តការហៅអនុគមន៍សម្រាប់ភ្នាក់ងារ អ្នកត្រូវការឲ្យមាន៖

១. ម៉ូឌែល LLM ដែលគាំទ្រការហៅអនុគមន៍
២. ស្គីមដែលមានការពិពណ៌នាអនុគមន៍
៣. កូដសម្រាប់អនុគមន៍នីមួយៗដែលបានពិពណ៌នា

យើងយកឧទាហរណ៍នៃការទទួលបានម៉ោងបច្ចុប្បន្ននៅក្នុងទីក្រុងមួយដើម្បីបង្ហាញ៖

១. **ផ្តើមម៉ូឌែល LLM ដែលគាំទ្រការហៅអនុគមន៍៖**

    មិនមូឌែលទាំងអស់គាំទ្រការហៅអនុគមន៍ទេ ដូច្នេះវាសំខាន់ក្នុងការត្រួតពិនិត្យថា LLM ដែលអ្នកកំពុងប្រើទើបនេះគាំទ្រមែនទេ។ <a href="https://learn.microsoft.com/azure/ai-services/openai/how-to/function-calling" target="_blank">Azure OpenAI</a> គាំទ្រការហៅអនុគមន៍។ យើងអាចចាប់ផ្តើមដោយបង្កើត OpenAI client ទៅប្រឆាំងនឹង Azure OpenAI **Responses API** (ចុងចំណុចដែលមានស្ថិរភាព `/openai/v1/` - មិនចាំបាច់ `api_version`)។

    ```python
    # ចាប់ផ្តើមម៉ាស៊ីនអតិថិជន OpenAI សម្រាប់ Azure OpenAI (API ជំនួស, ចំណុចធ្វើការ v1)
    client = OpenAI(
        base_url=f"{os.environ['AZURE_OPENAI_ENDPOINT'].rstrip('/')}/openai/v1/",
        api_key=os.environ["AZURE_OPENAI_API_KEY"],
    )
    deployment_name = os.environ["AZURE_OPENAI_DEPLOYMENT"]
    ```

១. **បង្កើតស្គីមអនុគមន៍**៖

    បន្ទាប់ យើងនឹងកំណត់ស្គីម JSON ដែលមានឈ្មោះអនុគមន៍ ការពិពណ៌នាអំពីអ្វីដែលអនុគមន៍ធ្វើ និងឈ្មោះជាមួយការពិពណ៌នាដើម្បីបញ្ជាក់ប៉ារ៉ាម៉ែត្រ។
    បន្ទាប់ យើងនឹងបញ្ជូនស្គីមនេះទៅ client ដែលបានបង្កើតមុន រួមជាមួយសំណើអ្នកប្រើ ដើម្បីស្វែងរកម៉ោងនៅ San Francisco។ អ្វីដែលសំខាន់គួរចំណាំ គឺថា **ការហៅឧបករណ៍** ជាអ្វីដែលត្រូវត្រឡប់មកវិញ មិនមែនចម្លើយចុងក្រោយនៃសំនួរនោះទេ។ ដូចបានបញ្ជាក់ពីមុន LLM ត្រឡប់ឈ្មោះនៃអនុគមន៍ដែលបានជ្រើសរើសសម្រាប់បេសកកម្ម និងអាគុយម៉ង់ដែលនឹងត្រូវផ្ញើទៅវា។

    ```python
    # ការពិពណ៌នាអំពីមុខងារសម្រាប់ម៉ូដែលក្នុងការអាន (ទ្រង់ទ្រាយឧបករណ៍ API ផ្តល់ចម្លើយរាបស)
    tools = [
        {
            "type": "function",
            "name": "get_current_time",
            "description": "Get the current time in a given location",
            "parameters": {
                "type": "object",
                "properties": {
                    "location": {
                        "type": "string",
                        "description": "The city name, e.g. San Francisco",
                    },
                },
                "required": ["location"],
            },
        }
    ]
    ```
   
    ```python
  
    # ការបញ្ជូនសារដំបូងពីអ្នកប្រើប្រាស់
    messages = [{"role": "user", "content": "What's the current time in San Francisco"}]

    # ការហៅ API ដំបូង៖ សុំម៉ូដែលប្រើមុខងារ
    response = client.responses.create(
        model=deployment_name,
        input=messages,
        tools=tools,
        tool_choice="auto",
        store=False,
    )

    # API ពីរបៅផ្តល់ឲ្យការហៅឧបករណ៍ជា function_call ក្នុង response.output។
    # បន្ថែមពួកវាទៅក្នុងការសន្ទនាដើម្បីឲ្យម៉ូដែលមានបរិបទពេញលេញនៅជំហានបន្ទាប់។
    messages += response.output

    print("Model's response:")
    print(response.output)
  
    ```

    ```bash
    Model's response:
    [ResponseFunctionToolCall(arguments='{"location":"San Francisco"}', call_id='call_pOsKdUlqvdyttYB67MOj434b', name='get_current_time', type='function_call')]
    ```
  
១. **កូដអនុគមន៍ដែលចាំបាច់សម្រាប់អនុវត្តបេសកកម្ម៖**

    ឥឡូវនេះពេល LLM បានជ្រើសរើសអនុគមន៍ដែលត្រូវដំណើរការ កូដដែលអនុវត្តបេសកកម្មត្រូវបានអនុវត្តន៍។
    យើងអាចសរសេរកូដក្នុងភាសា Python ដើម្បីរកម៉ោងបច្ចុប្បន្ន។ យើងត្រូវការសរសេរកូដខ្លះដើម្បីត្រួតពិនិត្យឈ្មោះ និងអាគុយម៉ង់ពី response_message ដើម្បីទទួលបានលទ្ធផលចុងក្រោយ។

    ```python
      def get_current_time(location):
        """Get the current time for a given location"""
        print(f"get_current_time called with location: {location}")  
        location_lower = location.lower()
        
        for key, timezone in TIMEZONE_DATA.items():
            if key in location_lower:
                print(f"Timezone found for {key}")  
                current_time = datetime.now(ZoneInfo(timezone)).strftime("%I:%M %p")
                return json.dumps({
                    "location": location,
                    "current_time": current_time
                })
      
        print(f"No timezone data found for {location_lower}")  
        return json.dumps({"location": location, "current_time": "unknown"})
    ```

     ```python
    # ឆែកការហៅមុខងារ
    tool_calls = [item for item in response.output if item.type == "function_call"]
    if tool_calls:
        for tool_call in tool_calls:
            if tool_call.name == "get_current_time":

                function_args = json.loads(tool_call.arguments)

                time_response = get_current_time(
                    location=function_args.get("location")
                )

                # ជម្លោះលទ្ធផលឧបករណ៍ជាធាតុ function_call_output
                messages.append({
                    "type": "function_call_output",
                    "call_id": tool_call.call_id,
                    "output": time_response,
                })
    else:
        print("No tool calls were made by the model.")

    # ការហៅ API ទីពីរ៖ សូមទទួលបានចម្លើយចុងក្រោយពីម៉ូដែល
    final_response = client.responses.create(
        model=deployment_name,
        input=messages,
        tools=tools,
        store=False,
    )

    return final_response.output_text
     ```

      The current time in San Francisco is 09:24 AM.

## ឧទាហរណ៍នៃការប្រើប្រាស់ឧបករណ៍ជាមួយស៊ុម Agentic

ឧទាហរណ៍ខាងក្រោមបង្ហាញពីរបៀបដែលអ្នកអាចអនុវត្តលំនាំរចនាប្រើប្រាស់ឧបករណ៍ដោយប្រើស៊ុម agentic ផ្សេងៗគ្នា៖

### ស៊ុម Microsoft Agent Framework

<a href="https://learn.microsoft.com/azure/ai-services/agents/overview" target="_blank">Microsoft Agent Framework</a> គឺជាស៊ុម AI មានកូដបើកចំហសម្រាប់បង្កើតភ្នាក់ងារប្រើប្រាស់ AI។ វាទាក់ទាញដំណើរការប្រើប្រាស់ការហៅអនុគមន៍ដោយអនុញ្ញាតឲ្យអ្នកកំណត់ឧបករណ៍ជាអនុគមន៍ Python ដោយប្រើតួ @tool។ ស៊ុមនេះគ្រប់គ្រងការទំនាក់ទំនងចេញ-ចូលរវាងម៉ូឌែល និងកូដរបស់អ្នកផងដែរ។ វាបន្ថែមការចូលប្រើឧបករណ៍ដែលបានបង្កើតជាមុនដូចជាការស្វែងរកឯកសារ និង Code Interpreter តាមរយៈ FoundryChatClient។

រាងគំនូរខាងក្រោមបង្ហាញដំណើរការនៃការហៅអនុគមន៍ជាមួយ Microsoft Agent Framework៖

![function calling](../../../translated_images/km/functioncalling-diagram.a84006fc287f6014.webp)

នៅក្នុង Microsoft Agent Framework ឧបករណ៍ត្រូវបានកំណត់ជាអនុគមន៍ដែលមានតួ @tool។ យើងអាចបំលែងអនុគមន៍ `get_current_time` ដែលបានឃើញពីមុនជាឧបករណ៍ដោយប្រើតួ @tool។ ស៊ុមនឹងស្វ័យប្រវត្តិ serialize អនុគមន៍ និងប៉ារ៉ាម៉ែត្ររបស់វា បង្កើតស្គីមសម្រាប់ផ្ញើទៅ LLM។

```python
import os
from agent_framework import tool
from agent_framework.foundry import FoundryChatClient
from azure.identity import AzureCliCredential

@tool(approval_mode="never_require")
def get_current_time(location: str) -> str:
    """Get the current time for a given location"""
    ...

# បង្កើតអតិថិជន
provider = FoundryChatClient(
    project_endpoint=os.environ["AZURE_AI_PROJECT_ENDPOINT"],
    model=os.environ["AZURE_AI_MODEL_DEPLOYMENT_NAME"],
    credential=AzureCliCredential(),
)

# បង្កើតភ្នាក់ងារ និងរត់ជាមួយឧបករណ៍
agent = provider.as_agent(name="TimeAgent", instructions="Use available tools to answer questions.", tools=get_current_time)
response = await agent.run("What time is it?")
```
  
### សេវា Microsoft Foundry Agent

<a href="https://learn.microsoft.com/azure/ai-services/agents/overview" target="_blank">Microsoft Foundry Agent Service</a> គឺជាស៊ុម agentic ថ្មីដែលរចនាឡើងដើម្បីផ្ដល់អំណាចដល់អ្នកអភិវឌ្ឍសម្រាប់បង្កើត បញ្ចេញ និងវាស់វែងភ្នាក់ងារប្រើប្រាស់ AI ល្អប្រសើរ និងអាចបន្ថែមបាន ដោយមិនចាំបាច់គ្រប់គ្រងធនធានគណនានិងផ្ទុកដែលនៅពីក្រោយ។ វាអាចប្រើប្រាស់សម្រាប់កម្មវិធីសហគ្រាសព្រោះជាសេវាប្រភេទគ្រប់គ្រងពេញលេញជាមួយសុវត្ថិភាពថ្នាក់សហគ្រាស។

ក្រោមការប្រៀបធៀបជាមួយការអភិវឌ្ឍដោយប្រើ API LLM ផ្ទាល់ Microsoft Foundry Agent Service ផ្តល់អត្ថប្រយោជន៍ខ្លះៗ រួមមាន៖

- ការហៅឧបករណ៍ស្វ័យប្រវត្តិ – មិនចាំបាច់បំលែងការហៅ ហៅឧបករណ៍, និងគ្រប់គ្រងចម្លើយឡើយ; រឿងទាំងនេះត្រូវបានបំពេញនៅជាសេវា
- ទិន្នន័យគ្រប់គ្រងយ៉ាងសុវត្ថិភាព – ជំនួសការគ្រប់គ្រងស្ថានភាពសន្ទនារបស់ខ្លួន អ្នកអាចពឹងផ្អែកលើ threads ដើម្បីរក្សាទុកព័ត៌មានទាំងអស់ដែលអ្នកចាំបាច់
- ឧបករណ៍មានរួចពីរ – ឧបករណ៍ដែលអ្នកអាចប្រើសម្រាប់អន្តរកម្មជាមួយប្រភពទិន្នន័យរបស់អ្នក ដូចជា Bing, Azure AI Search, និង Azure Functions។

ឧបករណ៍ដែលមាននៅក្នុង Microsoft Foundry Agent Service អាចបែងចែកជាពីរប្រភេទ៖

    - <a href="https://learn.microsoft.com/azure/ai-services/agents/how-to/tools/file-search?tabs=python&pivots=overview" target="_blank">ការស្វែងរកឯកសារ</a>

សេវាAgent អនុញ្ញាតឲ្យយើងប្រើឧបករណ៍ទាំងនេះរួមគ្នាជា `toolset`។ វាក៏ប្រើប្រាស់ `threads` ដែលតាមដានប្រវត្តិសារៗពីសន្ទនាផ្សព្វផ្សាយជាក់លាក់។

សូមរំលឹកថាអ្នកជាភ្នាក់ងារលក់នៅក្រុមហ៊ុនមួយឈ្មោះ Contoso។ អ្នកចង់បង្កើតភ្នាក់ងារសន្ទនាដែលអាចឆ្លើយសំណួរអំពីទិន្នន័យលក់របស់អ្នក។

រូបភាពខាងក្រោមបង្ហាញពីរបៀបដែលអ្នកអាចប្រើ Microsoft Foundry Agent Service ដើម្បី វិភាគទិន្នន័យលក់របស់អ្នក៖

![Agentic Service In Action](../../../translated_images/km/agent-service-in-action.34fb465c9a84659e.webp)

ដើម្បីប្រើឧបករណ៍ណាមួយជាមួយសេវា យើងអាចបង្កើត client និងកំណត់ឧបករណ៍ ឬ toolset ។ ដើម្បីអនុវត្តយ៉ាងផែនការ យើងប្រើកូដ Python ខាងក្រោម។ LLM អាចមើលទៅ toolset ហើយសម្រេចចិត្តថាតើប្រើអនុគមន៍ do-it-yourself `fetch_sales_data_using_sqlite_query` ឬ Code Interpreter ដែលបានបង្កើតជាមុន សម្រាប់ទៅសំណើអ្នកប្រើ។

```python 
import os
from azure.ai.projects import AIProjectClient
from azure.identity import DefaultAzureCredential
from fetch_sales_data_functions import fetch_sales_data_using_sqlite_query # មុខងារ fetch_sales_data_using_sqlite_query ដែលអាចរកបានក្នុងឯកសារ fetch_sales_data_functions.py ។
from azure.ai.projects.models import ToolSet, FunctionTool, CodeInterpreterTool

project_client = AIProjectClient.from_connection_string(
    credential=DefaultAzureCredential(),
    conn_str=os.environ["PROJECT_CONNECTION_STRING"],
)

# ចាប់ផ្តើមបរិក្ខារ
toolset = ToolSet()

# ចាប់ផ្តើមភ្នាក់ងារហៅមុខងារជាមួយមុខងារ fetch_sales_data_using_sqlite_query ហើយបន្ថែមវាទៅក្នុងបរិក្ខារ
fetch_data_function = FunctionTool(fetch_sales_data_using_sqlite_query)
toolset.add(fetch_data_function)

# ចាប់ផ្តើមឧបករណ៍ Code Interpreter ហើយបន្ថែមវាទៅក្នុងបរិក្ខារ។
code_interpreter = CodeInterpreterTool()toolset.add(code_interpreter)

agent = project_client.agents.create_agent(
    model="gpt-4.1-mini", name="my-agent", instructions="You are helpful agent", 
    toolset=toolset
)
```

## តើមានការពិចារណាពិសេសអ្វីខ្លះសម្រាប់ប្រើលំនាំរចនាប្រើប្រាស់ឧបករណ៍ដើម្បីបង្កើតភ្នាក់ងារប្រើប្រាស់ AI ដែលអាចទុកចិត្តបាន?

ប្រសាសន៍ធម្មតាមួយស្ដីពី SQL ដែលបានបង្កើតដោយម៉ូឌែល LLM មានការព្រួយបារម្ភវត្ថុសន្តិសុខ ជាពិសេសការជំរុញ SQL injection ឬសកម្មភាពគំរាមកំហែង ដូចជាការរំលង ឬកែប្រែទិន្នន័យលើតារាងទិន្នន័យ។ ទោះជាការព្រួយបារម្ភទាំងនេះជាការពិត ក៏ប៉ុន្តែអាចបញ្ច្រាសបានយ៉ាងមានប្រសិទ្ធភាពដោយការកំណត់សិទ្ធិចូលប្រើតារាងទិន្នន័យឲ្យត្រឹមត្រូវ។ សម្រាប់តារាងទិន្នន័យភាគច្រើន នេះមានន័យថាកំណត់ផ្ទាំងទិន្នន័យឲ្យអានបានតែប៉ុណ្ណោះ។ សម្រាប់សេវាកម្មដូចជា PostgreSQL ឬ Azure SQL អាចចាត់ចែងប្រតិបត្តិការឲ្យ app មានតួនាទីអានបានតែប៉ុណ្ណោះ (SELECT)។

ការរត់កម្មវិធីនៅក្នុងបរិយាកាសសុវត្ថិភាពឡើងវិញក៏ជួយបង្កើនការពារជាងមុន។ នៅក្នុងស្ថានការកម្មវិធីសហគ្រាស ទិន្នន័យភាគច្រើនត្រូវបានដក ហើយបម្លែងពីប្រព័ន្ធប្រតិបត្ដិការ ទៅក្នុងតារាងទិន្នន័យអានតែបាន ឬឃ្លាំងទិន្នន័យ ដែលមានស្គីមងាយប្រើ។ វិធីនេះធានាថាទិន្នន័យមានសុវត្ថិភាព ប្រសិទ្ធភាពល្អ និងអាចចូលដំណើរការបានយ៉ាងងាយស្រួល ហើយកម្មវិធីមានសិទ្ធិអានតែប៉ុណ្ណោះ។
